A program is a container for one or more user-facing components, to be surfaced in both Centrapay and third-party applications

## Program Model

<Properties>
  <Property name="id" type="string">
    The unique identifier of the Program.
  </Property>

  <Property name="name" type="string">
    The display name of the Program.
  </Property>

  <Property name="type" type="string">
    The type of functionality contained in the Program. Valid values: `loyalty`.
  </Property>

  <Property name="description" type="string">
    The displayable description for the Program.
  </Property>

  <Property name="mediaUploadId" type="string">
    The id of the [Media Upload](/api/media-uploads/) image of the Program.
  </Property>

   <Property name="img" type="string">
    The img URL of the Program.
  </Property>

  <Property name="accountId" type="string">
    The [Account](/api/accounts/) that will own the Program.
  </Property>

   <Property name="brandId" type="string">
    The id of the Brand that will be associated with the Program.
  </Property>

  <Property name="createdAt" type="timestamp">
    When the Program was created.
  </Property>

  <Property name="createdBy" type="crn">
    The User or API Key that created the Program.
  </Property>

  <Property name="test" type="boolean">
    `true` if the Program is for testing purposes only.
  </Property>
</Properties>

<Endpoint
  path="/api/programs"
  filename="programs-create"
>
  ## Create Program <Badge type="experimental"/>

  This endpoint allows you to create a [Program](#program-model).

  <Properties>
    <Property name="name" type="string" required>
     The display name of the Program.
    </Property>

    <Property name="type" type="string" required>
     The type of functionality contained in the Program. Valid values: `loyalty`.
    </Property>

    <Property name="accountId" type="string" required>
      The Centrapay accountId for an org account.
    </Property>

    <Property name="description" type="string">
      The displayable description for the Program.
    </Property>

    <Property name="mediaUploadId" type="string">
      The id of the [Media Upload](/api/media-uploads/) image of the Program.
    </Property>

    <Property name="brandId" type="string">
      The id of the Brand that will be associated with the Program.
    </Property>
  </Properties>
</Endpoint>
